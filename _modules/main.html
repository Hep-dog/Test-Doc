#!/usr/bin/python3
# -*- coding:utf-8 -*-

from Core.worker import Run
from multiprocessing import Event, Lock, Value, Process, Queue
import sys, os, time, traceback
from Core.runCtrl import parseRunParas, sigint_handle, processAsyner, initDataBusDict
from Core.aggregationFuncs import procAggregation, procAggredDataProcess

'''
    Executive function of the Kafka-DataProcess-Package
'''


[文档]def main():
    # Topics related parameters initialization from the input JSON file
    '''
        topic parameters dictionary:
        {
            TopicName       : xxx,    --> name to topic
            BrokerIP        : xxx:xx, --> IP and Port
            Group_ID        : xxx,    --> consumer group_id
            Detectorcode    : 201010, --> detector code for analysis
            DataType        : Event,  --> one of "EventData", "MetaData" and "RawData"
            MainApp         : xxx,    --> name of the main application ( only one )
            Apps            : dict for multi applications
                {
                    Normal  : normal apps list,
                    Aggre   : aggregation apps list,
                }
            OtherApps       : xxx,xxx --> name of other applications
        }

        applications parameters dictionary
        { "Name" : { "ModuleName": xxx, "ClassName": xxx } }
    '''
    topicsParasList = []
    consumerParasDict = {}
    consuDataPipeDict = { "outDataPipeDict":{} }
    aggreDataPipeDict = { "inDataPipeDict":{}, "outDataPipeDict":{} }
    analyDataPipeDict = { "inDataPipeDict":{}, "dfDealFuncDict" :{} }

    try:
        topicJson, consumerJson = sys.argv[1], sys.argv[2]
        topicsParasDict, consumerParasDict = parseRunParas(topicJson), parseRunParas(consumerJson)
        topicsParasList, appsParasDict = topicsParasDict['Topics'], topicsParasDict['Apps']
        initDataBusDict( topicsParasList, appsParasDict, consuDataPipeDict, aggreDataPipeDict, analyDataPipeDict )
    except:
        traceback.print_exc()
        print( "\033[41mError: failed in get the topic and consumer parameters from json files!\033[0m" )
        print( "\033[41mUsage: python3 main.py topicConfig.json consumerConfig.json\033[0m")
        sys.exit(1)

    glbCtrlDict = {}
    glbCtrlDict['event'] = Event()
    glbCtrlDict['lock']  = Lock()
    glbCtrlDict['queue'] = Queue()
    glbCtrlDict['asynCNT'] = Value('i',0)
    glbCtrlDict['progressCNT'] = Value('i',0)

    procsList = []


    for eachTopicParasDict in topicsParasList:
        topicName = eachTopicParasDict["TopicName"]
        #singleTopicAppsList = sum( list(eachTopicParasDict['Apps'].values()), [] )
        procsList.append( Run(  eachTopicParasDict, consumerParasDict, glbCtrlDict, appsParasDict, consuDataPipeDict["outDataPipeDict"][topicName],)
                        )
    for aggreName in aggreDataPipeDict["inDataPipeDict"].keys():
        procsList.append( Process( target=procAggregation, args=( aggreDataPipeDict["inDataPipeDict"][aggreName], aggreDataPipeDict["outDataPipeDict"][aggreName], glbCtrlDict['queue'] ) ) )
        #procsList.append( Process( target=procAggredDataProcess,  args=( analyDataPipeDict["inDataPipeDict"][aggreName],  analyDataPipeDict["dfDealFuncDict"][aggreName], glbCtrlDict['queue'] ) ) )
        procsList.append( Process( target=procAggredDataProcess,  args=( analyDataPipeDict["inDataPipeDict"][aggreName],  appsParasDict[aggreName], glbCtrlDict['queue'] ) ) )

    # Create process for other processes controlling
    pCtl = Process( name='Controller', target=processAsyner, args=( len(topicsParasList), glbCtrlDict, sys.argv[1], sys.argv[2] ) )
    procsList.append( pCtl )

    for proc in procsList:
        proc.daemon = True
        proc.start()
    for proc in procsList:
        proc.join()

    print( "\033[42mINFO: Exiting the main process \033[0m" )


if __name__ == '__main__':
    main()
