import time, os, logging

[文档]class ChopperIDMonitor(MsgProcess):
    """
        Inherited from the template data process class "MsgProcess"
        This class is used to monitor the pulse ID's continuity of CSNS chopper MetaData
    """
    def __init__(self, appName):
        super().__init__(appName)
        self.recordFlag         = False
        self.logger             = None
        self.fileHandler        = None
        self.logEventLength     = 90000
        self.chopperIDGapLimit  = 1000
        self.outputDir          = 'output/'
        self.batchNumber        = 167
        self.idStatus           = 0
        self.eventStatusList    = []
        self.prePulseID         = 0
        self.preTenIDList       = []
        self.dataDict           = {}
        self.endPoint           = 'Jiyizi-test'
        self.metricHead         = 'PulseID-'
        self.pushAddress        = 'http://10.1.26.63:1988/v1/push'


[文档]    def run( self, msg, eventCounter, detectorTypeCode, dataType, dataPipeDict ):
        buf = msg.value
        data = MetaData.MetaData.GetRootAsMetaData( buf, 0 )
        ID = data.PulseId()
        deviceName = data.DeviceName()
        if deviceName not in self.dataDict:
            self.idStatus = 0
            self.dataDict[deviceName] = ID
        else:
            self.idStatus = ID - self.dataDict[deviceName]
            self.dataDict[deviceName] = ID

        PulseIDStatus = -1

        self.eventStatusList.append( self.idStatus )
        self.prePulseID = ID

        if( len(self.preTenIDList) ) > 10:
            self.preTenIDList.pop(0)

        self.preTenIDList.append( self.prePulseID )

        # ============== Setup For the Logger to record wrong events =======================================
        if (eventCounter)%self.logFileSize == 1:
            # check the data path here
            dataDir = self.outputDir + msg.topic + "/"
            if not os.path.exists( dataDir ):
                os.makedirs( dataDir )

            currentTime = time.strftime("%Y-%m-%d-%H-%M", time.localtime())
            outputFile = dataDir + 'Log_' + currentTime

            self.setupUserLogger( msg.topic, outputFile )
            print( "    ++++++++++++++++++++++ New Online PulseID Monitor Logger has been created +++++++++++++    " )

        #  ========================== Recording the logger ==================================
        if (self.idStatus<0) or (self.idStatus>self.chopperIDGapLimit):
                self.logger.info( ' Pulse ID error   , previous ten IDs: %120s,  current ID: %12s,  status: %s' \
                              %( self.preTenIDList, ID, self.idStatus  )  )
        # ===================================================================================================

        #  ========================== Push Data to OpenFalcon ===============================================
        if (eventCounter) % self.batchNumber == 0:
            targetList = []

            try:
                time_st = int(time.time())
                # Default is Pulse ID Monitor
                nameMetric = self.metricHead + msg.topic
                PulseIDStatus = np.mean( self.eventStatusList )
                self.collecteData( targetList, self.endPoint, nameMetric, PulseIDStatus, time_st )

                # Push all data to OpenFalcon
                roll = self.requestPV( targetList )
                print( "ErrorCounting", nameMetric, PulseIDStatus, time_st )
                print( "\033[42mINFO: %s requests sending status: %s\033[0m" % ( msg.topic, roll.text ) )

            except:
                traceback.print_exc()
                exit(1)

            # clear the Status List for next batch
            self.eventStatusList.clear()
        # ===================================================================================================

        if self.LogRecordFlag is True:
            if (eventCounter)%self.logFileSize == 0:
                self.closeLogger()

        if (eventCounter)%10000 == 1:
            print( " "*10, "Topic: %45s,\t ID: %10d,\t Offset: %8d "%(msg.topic, ID, msg.offset)  )
