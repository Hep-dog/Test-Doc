        """
            Template function used to format data for further push to OpenFalcon platform\n
            Args:\n
                resultList : list holds records of user target visualization results\n
                endPoint (str) : name of endPoint for OpenFalcon\n
                Metric (str) : name of metric for OpenFalcon\n
                valaue (data): user data processing results\n
                time_st      : time stamp for OpenFalcon\n
        """
        record = {}
        record['endPoint']  = endPoint
        record['step'] = 30
        record['counterType'] = 'GAUGE'
        record['metric'] = Metric
        record['value']  = value
        record['timestamp'] = time_st
        resultList.append( record )

[文档]    def setupUserLogger( self, nameTopic, outputFile ):
        """
            Template python logger setting function.\n
            Args:\n
                nameTopic (str) : name of the topic\n
                outputFile (str): name of the output file
        """
        level = logging.INFO
        formatter = logging.Formatter( '%(asctime)s:    %(message)s' )
        handler   = logging.FileHandler( outputFile )
        handler.setFormatter( formatter )
        logger = logging.getLogger( nameTopic )
        logger.setLevel( level )
        logger.addHandler( handler )
        self.logger = logger
        self.fileHandler = handler



[文档]    def closeLogger(self):

            Function used to close the logger.
        """
        self.logger.removeHandler( self.fileHandler )
        self.logger = None
        self.fileHandler = None
