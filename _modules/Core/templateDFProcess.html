[文档]    def collectData( self, resultList, endPoint, Metric, value, time_st ):
        """
            Template function used to format data for further push to OpenFalcon platform\n
            Args:\n
                resultList : list holds records of user target visualization results\n
                endPoint (str) : name of endPoint for OpenFalcon\n
                Metric (str) : name of metric for OpenFalcon\n
                valaue (data): user data processing results\n
                time_st      : time stamp for OpenFalcon\n
        """
        record = {}
        record['endPoint']      = endPoint
        record['step']          = 30
        record['counterType']   = 'GAUGE'
        record['metric']        = Metric
        record['value']         = value
        record['timestamp']     = time_st
        resultList.append( record )



[文档]    def requestPV( self, result ):
        """
            Template function used to push collect records to OpenFalcon\n
            Args:
                Remove IP: which is defined in the initialization of the class
                result   : the collected list of data records
        """
        roll = requests.post( self.pushAddress, data=json.dumps(result) )
        return roll
