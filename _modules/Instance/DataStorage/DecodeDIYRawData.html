[文档]def decodeMetaData():
    """
        This function is used to decode the DIY binary meta data.

        Note: There is an addition 4B int data head of each Pulse, which is the event
            length of these pulse.
    """
    if len(sys.argv) != 2:
        print( " =================== Usage: python3 Read.py fileName ===================" )
        sys.exit(1)
    else:
        dataName = sys.argv[1]

    f = open( dataName, 'rb' )
    size = os.path.getsize( dataName )
    print( size )

    tagSize = struct.calcsize("I")
    i = 0

    while i< size:
        data = f.read(tagSize)
        ( elength, ) = struct.unpack("I", data)
        buf = f.read(elength)
        data = MetaData.MetaData.GetRootAsMetaData( buf, 0)
        ID = data.PulseId()
        valueVec = data.ValueAsNumpy()
        deviceID = data.DeviceId()
        deviceName=data.DeviceName()
        print( " PulseID: %s, valueList: %s, deviceID: %5s, deviceName: %s " \
              % ( ID, list(valueVec), deviceID, deviceName ) )

        i += (tagSize+elength)
