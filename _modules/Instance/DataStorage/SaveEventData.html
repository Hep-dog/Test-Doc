[文档]    def createH5File(self, msg, pulseInfo ):
        """
            Creating the files
        """
        # Create the data file firstly
        dataDir = self.dataPath + msg.topic + "/" + self.subDirName + "/"
        if not os.path.exists( dataDir ):
            os.makedirs( dataDir )
        self.dataFileName   = dataDir + msg.topic + "_ID_" + str( pulseInfo[0] ) + "-" + str( pulseInfo[0] + int(self.dataLength-1)) + ".h5"
        self.dataFile       = h5py.File( self.dataFileName, 'w' )

        group   = self.dataFile.create_group( self.groupName )
        dbTof   = self.dataFile.create_dataset( self.tofDSName  , (pulseInfo[1],), compression='gzip', compression_opts=9, maxshape=(None,), dtype='int32' )
        dbPixel = self.dataFile.create_dataset( self.pixelDSName, (pulseInfo[1],), compression='gzip', compression_opts=9, maxshape=(None,), dtype='int32' )
        dbEventInfo = self.dataFile.create_dataset( self.eventDSName, (1,3)      , compression='gzip', compression_opts=9, maxshape=(None,3), dtype='int64' )
        print("\033[42mINFO: Create data HDF5 file: %s\033[0m" % self.dataFileName )



[文档]    def saveToH5File( self, pulseInfo, tofVec, pixelVec ):
        """
            Performing the data storage.
        """
        self.dataFile = h5py.File( self.dataFileName, 'a' )
        tofDS = self.dataFile[ self.tofDSName ]
        eventDS = self.dataFile[ self.eventDSName ]
        pixelDS = self.dataFile[ self.pixelDSName ]

        tofDS.resize(   [ self.neutronCounter + pulseInfo[1] ] )
        pixelDS.resize( [ self.neutronCounter + pulseInfo[1] ] )
        eventDS.resize( [ self.fileProcessCounter+1, 3 ] )

        tofDS[    self.neutronCounter : self.neutronCounter+pulseInfo[1] ] = tofVec
        pixelDS[  self.neutronCounter : self.neutronCounter+pulseInfo[1] ] = pixelVec

        self.neutronCounter = self.neutronCounter + pulseInfo[1]
        pulseInforNP = np.array( [ pulseInfo[0],  pulseInfo[1], self.neutronCounter] )
        eventDS[ self.fileProcessCounter : self.fileProcessCounter+1 ] = pulseInforNP

        self.fileProcessCounter += 1
        #print (  self.fileProcessCounter,  self.dataLength   )
        if self.fileProcessCounter == ( self.dataLength ):
            self.dataFile.close()
            self.neutronCounter = 0
            self.fileProcessCounter = 0



[文档]    def run( self, msg, eventCounter, detectorTypeCode, dataType, dataPipeDict ):
        buf = msg.value
        data = EventData.EventData.GetRootAsEventData( buf, 0 )
        tofVec = data.TofAsNumpy()
        pixelVec = data.PosAsNumpy()
        eventLen = data.TofLength()
        ID = data.PulseId()

        pulseInfo = [ ID, eventLen, [self.neutronCounter] ]
        if (self.fileProcessCounter) % (self.dataLength) == 0:
            self.createH5File( msg, pulseInfo )
        self.saveToH5File( pulseInfo, tofVec, pixelVec )

        if eventCounter%1000 == 1:
            print( " "*10, "Topic: %46s,\t ID: %10d,\t Offset: %8d "%(msg.topic, ID, msg.offset)  )
