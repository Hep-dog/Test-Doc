#!/usr/bin/python

import os
import struct
from fbs.Data.Detector import EventData
from Core.templateMsgProcess import MsgProcess


[文档]class SaveDIYRawData(MsgProcess):
    """
        Inherited from the template data process class "MsgProcess"
        This class is used to save the CSNS eventdata data, to the DIY binary format.

        Note: The event length( 4B size ) is taken as an additional data head every pulse data, which is
        used to for the data splitting in further binary data decoding.
    """
    def __init__(self, appName):
        super().__init__(appName)
        self.subDirName = "RawData"
        self.dataLength = 10000
        self.dataFile   = None
        self.dataPath   = "output/"
        self.dataFileName = ""
        self.fileProcessCounter = 0


[文档]    def finalize( self, topicParasDict ):
        """
            Closing the files
        """
        self.dataFile.close()



[文档]    def createBinFile( self, msg, ID ):
        """
            Creating the files
        """
        # Create the data file firstly
        # Check the data path is valid or not
        dataDir = self.dataPath + "/" + msg.topic + "/" + self.subDirName + "/"
        if not os.path.exists( dataDir ):
            os.makedirs( dataDir )

        self.dataFileName = dataDir + msg.topic + "-ID" + str( ID ) + ".bin"
        self.dataFile = open(str( self.dataFileName ), "wb")
        print("\033[42mINFO: Create data Bin file: %s\033[0m" % self.dataFileName )



[文档]    def saveBinData( self, buf, cnt ):
        """
            Perform the data storage.
        """
        stream = bytes(buf)
        transBuf = struct.pack("I%ds" % (len(stream),), len(stream), stream)
        self.dataFile.write( transBuf )
        self.fileProcessCounter += 1

        if self.fileProcessCounter == ( self.dataLength ):
            self.dataFile.close()
            self.fileProcessCounter = 0



[文档]    def run( self, msg, eventCounter, detectorTypeCode, dataType, dataPipeDict ):
        buf = msg.value
        data = EventData.EventData.GetRootAsEventData( buf, 0 )
        ID = data.PulseId()

        if (self.fileProcessCounter) % (self.dataLength) == 0:
            self.createBinFile( msg, ID )
        self.saveBinData( buf, eventCounter )

        if eventCounter%10000 == 1:
            print( " "*10, "Topic: %46s,\t ID: %10d,\t Offset: %8d "%(msg.topic, ID, msg.offset)   )
