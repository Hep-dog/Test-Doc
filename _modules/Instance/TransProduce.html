#!/usr/bin/python

import sys, traceback
from kafka import KafkaProducer
from fbs.Data.Detector import EventData
from Core.templateMsgProcess import MsgProcess


[文档]class TransProduce(MsgProcess):
    """
        Inherited from the template data process class "MsgProcess"
        This class is used to Transfer the data in first Kafka ring
        of CSNS, to the second one.
    """
    def __init__(self, appName):
        super().__init__(appName)
        self.producer   = None
        self.producerIP = 'localhost:9092'
        self.additionName= ''


[文档]    def createProducer(self, topicParasDict):
        """
            Creating the producer
        """
        self.producer = KafkaProducer( bootstrap_servers=self.producerIP,
                                  max_request_size=1048576,
                                  batch_size=1048576,
                                  retries=5,
                                 )
        print("\033[42mINFO: User producer has been created for topic: %s\033[0m" % topicParasDict['TopicName'] )



[文档]    def initialize(self, topicParasDict):
        """
            Creating the producer before the event loop
        """
        self.createProducer(topicParasDict)



[文档]    def finalize(self, topicParasDict):
        """
            Flush and close the Kafka producer instance.
        """
        if self.producer:
            self.producer.flush()
            self.producer.close()
            print("\033[42mINFO: User producer has been closed for topic: %s\033[0m" % topicParasDict['TopicName'] )



[文档]    def run( self, msg, eventCounter, detectorTypeCode, dataType, dataPipeDict ):
        buf = msg.value
        data = EventData.EventData.GetRootAsEventData( buf, 0 )
        ID = data.PulseId()

        try:
            self.producer.send( (self.additionName+msg.topic), buf  )
        except:
            print( "\033[41ERROR: %s sending data error!\033[0m" % msg.topic )
            traceback.print_exc()
            sys.exit(1)

        if eventCounter%1000 == 0:
            print( " "*10, "Topic: %46s,\t ID: %10d,\t Offset: %8d "%(msg.topic, ID, msg.offset) )
