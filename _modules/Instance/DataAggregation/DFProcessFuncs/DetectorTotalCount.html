[文档]def getSecAllCounting( dataDict, protonFre ):
    secCounting = []
    secIDList = []
    signalList   = dataDict["leftData"]   + dataDict["dataList"]
    signalIDList = dataDict["leftIDList"] + dataDict["idList"]
    dataDict["leftData"].clear()
    dataDict["leftIDList"].clear()

    for i in range(len(signalList)%protonFre):
        dataDict["leftData"]  .insert( 0, signalList.pop() )
        dataDict["leftIDList"].insert( 0, signalIDList.pop() )
    if len(signalList)/protonFre>=1:
        secCounting = list( np.sum( (np.array(signalList)).reshape(-1,protonFre), axis=1  ) )
        secIDList   = (np.array(signalIDList).reshape(-1,protonFre))[:,0].tolist()
    return secCounting, secIDList
