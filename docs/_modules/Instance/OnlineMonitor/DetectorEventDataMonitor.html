#!/usr/bin/python

import sys,time, os
import traceback
from fbs.Data.Detector import EventData
from Core.templateMsgProcess import MsgProcess


[文档]class DetectorEventDataMonitor(MsgProcess):
    """
        Inherited from the template data process class "MsgProcess"
        This class is used to decode the CSNS detector EventData, and display them in the online website,
        including:\n
            1) Pulse ID's continuity. \n
            2) Tof Range (0-40)ms, the arbitary scale factor corresponding with detector should be taken as ana
                input parameter "tofUnit".\n
            3) Pixel Range (32,32) for monitor and scintillator, (1-60) for short 3He tube, and (1-100) for long one.\n
    """
    def __init__(self, appName):
        super().__init__(appName)
        self.LogRecordFlag  = True
        self.logger         = None
        self.fileHandler    = None
        self.logFileSize    = 2160000
        self.outputDir      = 'output/'
        self.batchNumber    = 750
        self.idStatus       = 0
        self.idStatusList   = []
        self.prePulseID     = -1
        self.invalidID      = 999999999
        self.preTenIDList   = []
        self.endPoint       = "Test-BL20-"
        self.metricHead     = ""
        self.pushAddress    = 'http://10.1.26.63:1988/v1/push'
        self.detectorType   = 0 # 0 for He3, 1 for Monitor, 2 for Scin
        self.subDetectorType= 0 # He3: 0 for long tube and 1 for short, Scin: 0 and 1 are same,
        self.pixelRange     = ( (800,480), (32,32), (32,32) ) # long tube: 100*8, short tube: 60*8
        self.tofFactor      = ( 1, 1, 8 ) # 1 for He3 and Monitor, 8 for Scin
        self.tofUnit        = ( 1, 1000, 1000000 ) # ms for He3 and Monitor, ns for Scin
        self.tofRanges      = (0, 40)
        self.pixelStatusList= []
        self.tofStatusList  = []

        if int(detectorTypeCode)/100000<0 or int(detectorTypeCode)/100000>10:
            print( "\033[41mERROR: Topic: %s invalid input Detector Type Code, exist the program!\033[0m" % topicName )
            sys.exit(1)
        self.detectorType = int(int(detectorTypeCode)/100000)
        self.subDetectorType= int( (int(detectorTypeCode)%10) )
        print( "\033[42mINFO: Topic: %s Detector type: %s, subDetectorType: %s, pixel Range: %s\033[0m" \
            % ( topicName, self.detectorType, self.subDetectorType, self.pixelRange[self.detectorType-1][self.subDetectorType-1] ) )

[文档]    def run( self, msg, eventCounter, detectorTypeCode, dataType, dataPipeDict ):
        buf = msg.value
        data = EventData.EventData.GetRootAsEventData( buf, 0 )
        tofVec, posVec, eventLen, ID = data.TofAsNumpy(), data.PosAsNumpy(), data.TofLength(), data.PulseId()
        PulseIDStatus = -1

        pixelList= None
        errPixelCNT= 0
        errTofCNT = 0

        # ==================   For the tof data monitor ==========================
        try:
            transTofVec = tofVec  \
                        * ( (self.tofFactor)[(self.detectorType)] ) \
                        / ( (self.tofUnit)[self.detectorType] )
            errTofCNT   += len(transTofVec[ transTofVec < (self.tofRanges)[0] ])
            errTofCNT   += len(transTofVec[ transTofVec > (self.tofRanges)[1] ])
            self.tofStatusList.append( errTofCNT )
        except:
            print( "\033[41mERROR: Topic: %s decording the Tof vec error\033[0m" % msg.topic )
            traceback.print_exc()
            sys.exit(1)

        # ==================   For the pixel data monitor ==========================
        #Decode the bank, module and pixel
        if eventLen>0:
            pixelList= ( (posVec)%100 )
            errPixelCNT += len(pixelList[ pixelList < 0])
            errPixelCNT += len(pixelList[ pixelList > ( ((self.pixelRange)[self.detectorType-1])[self.subDetectorType-1]) ])
            self.pixelStatusList.append( errPixelCNT )

        # ==================   For the Pulse ID monitor ==========================
        if ID<self.invalidID and ID >0:
            if (self.prePulseID<0) or (ID-self.prePulseID == 1) :
                self.idStatus = 0
            else:
                self.idStatus = 1
        else:
            ID = self.invalidID
            self.idStatus = 1

        self.idStatusList.append( self.idStatus )
        self.prePulseID = ID

        if( len(self.preTenIDList) ) > 10:
            self.preTenIDList.pop(0)

        self.preTenIDList.append( self.prePulseID )

        # ============== Setup For the Logger to record wrong events =======================================
        if (eventCounter)%self.logFileSize == 1:
            # check the data path here
            dataDir = self.outputDir + msg.topic + "/"
            if not os.path.exists( dataDir ):
                os.makedirs( dataDir )

            currentTime = time.strftime("%Y-%m-%d-%H-%M", time.localtime())
            outputFile = dataDir + 'Log_' + currentTime

            self.setupUserLogger( msg.topic, outputFile )

        #  ========================== Recording the logger ==================================
        if self.idStatus != 0:
                self.logger.info( ' Pulse ID error   , previous ten IDs: %120s,  current ID: %12s,  status: %s' \
                              %( self.preTenIDList, ID, self.idStatus  )  )
        if errPixelCNT != 0:
                self.logger.info( ' Pixel range error, previous ten IDs: %120s,  current ID: %12s,  status: %s' \
                              %( self.preTenIDList, ID, errPixelCNT  )  )
        if errTofCNT != 0:
                self.logger.info( ' Tof range error, previous ten IDs: %120s,  current ID: %12s,  status: %s' \
                              %( self.preTenIDList, ID, errTofCNT  )  )

        #  ========================== Push Data to OpenFalcon ===============================================
        if len(self.idStatusList) == self.batchNumber:
            targetList = []
            try:
                time_st = int(time.time())
                nameMetric = self.metricHead + "PulseID-" + msg.topic
                PulseIDStatus = sum( self.idStatusList )
                self.collecteData( targetList, self.endPoint, nameMetric, PulseIDStatus, time_st )

                # Now for pixel monitoring
                pixelMetric= self.metricHead + "-PixelErrCNT-" + msg.topic
                PixelStatus= sum( self.pixelStatusList )
                self.collecteData( targetList, self.endPoint, pixelMetric, PixelStatus, time_st )

                # Now for Tof ranges Monitoring
                tofMetric= self.metricHead + "-TofErrCNT-" + msg.topic
                ToFStatus= sum( self.tofStatusList )
                self.collecteData( targetList, self.endPoint, tofMetric , ToFStatus , time_st )

                # Push all data to OpenFalcon
                roll = self.requestPV( targetList )
                print( " "*10, "ErrorCounting", nameMetric, PulseIDStatus, time_st  )
                print( " "*10, "Topic: %45s requests sending status: %s" % ( msg.topic, roll.text  )  )
            except:
                print( "\033[41mERROR: Topic: %s monitoring and displaying ID, pixel and Tof data occurs error!\033\[0m" % msg.topic )
                traceback.print_exc()
                exit(1)

            # clear the Status List for next batch
            self.idStatusList.clear()
            self.tofStatusList.clear()
            self.pixelStatusList.clear()
        # ===================================================================================================

        if (eventCounter)%self.logFileSize == 0:
            self.closeLogger()

        if eventCounter%10 == 1:
            print( " "*10, "Topic: %45s,\t ID: %10d,\t Offset: %8d "%(msg.topic, ID, msg.offset)   )
