[文档]    def collectEventDataCounting( self, msg ):
        """
            Collecting and storing the pulse ID and event rate of EventData.
        """
        buf = msg.value
        data= EventData.EventData.GetRootAsEventData( buf, 0 )
        self.dataDict["PulseID"].append(data.PulseId())
        self.dataDict[msg.topic].append(data.TofLength())



[文档]    def collectMetaDataCounting( self, msg ):
        """
            Collecting and storing the pulse ID and event rate of MetaData.
        """
        buf = msg.value
        data= MetaData.MetaData.GetRootAsMetaData( buf, 0 )
        self.dataDict["PulseID"].append(data.PulseId())
        try:
            self.dataDict[msg.topic].append([list(data.ValueAsNumpy())[2], data.TimeSecondAsNumpy(), data.TimeNanoAsNumpy()])
        except:
            print( "\033[41mERROR: MetaData Value length Error for topic: %s\033[0m" % msg.topic )
            traceback.print_exc()
            self.dataDict[msg.topic].append(0)
