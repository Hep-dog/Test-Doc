from Core.templateMsgProcess import MsgProcess
from fbs.Data.Detector import EventData
import pandas as pd
import traceback


[文档]class AggregationBase(MsgProcess):
    """
        Inherited from the template data process class "MsgProcess"
        This class is the mother class used to collect the CSNS EventData/MetaData,
        convert user's target variables waited to be aggregated, to the Pandas dataframe format,
        and transfer the dataframe to the data aggregation process.
    """
    def __init__(self, appName):
        super().__init__(appName)
        self.pulseFre   = 25
        self.secFactor  = 40
        self.distDrop   = 5000
        self.dataDict   = {}
        self.leftDFLis  = []
        self.funcMap    = {}


[文档]    def getID(self, msg):
        """
            Getting the pulse ID from specific Pulse beam data

            Args:\n
                msg (binary) : kafka  message

            Return: pulse ID (int)
        """
        try:
            buf = msg.value
            data= EventData.EventData.GetRootAsEventData( buf, 0 )
            ID = data.PulseId()
        except:
            traceback.print_exc()
        return ID



[文档]    def collectMessage(self, msg):
        """
            Function used to collecte the target variable from each Pulse data, to the defined data dictionary.\n

            Note: This function should be override by user, in order to get the corresponding variabes
            in CSNS EventData/MetaData. ( User can define they functions alternatively, as shonw in the
            "AggreExample" sub-module.)
        """
        buf = msg.value
        data= EventData.EventData.GetRootAsEventData( buf, 0 )
        self.dataDict["PulseID"].append(data.PulseId())
        self.dataDict[msg.topic].append(msg)




[文档]    def initialize( self, topicParasDict ):
        """
            Initializing the function map and dataDict, to decode and storage the data.

            Note: User should override this function corresponding the target data variable,
            i.e., for scale monitor event counting and proton charge, here should be two decoding
            function maps in the "funcMap" dict, which are used to collect and storage the two
            data source respectively, as shown in the "AggreExample" sub-module.
        """
        self.funcMap    = { 'EventData':self.collectMessage }
        self.dataDict = { "PulseID":[], topicParasDict["TopicName"]:[] }

[文档]    def run( self, msg, eventCounter, detectorTypeCode, dataType, dataPipeDict ):

        self.funcMap[dataType]( msg )

        if (eventCounter) % (self.pulseFre*self.secFactor) == 0:
            currentDF = pd.DataFrame.from_dict( self.dataDict )
            currentDF = currentDF.set_index('PulseID')
            print( " "*10, "Topic: %45s,\t ID: %10d,\t eventCNT: %8d "%(msg.topic, self.getID(msg), eventCounter)  )

            dataPipeDict[self.appName].send(currentDF)
            _ = [ self.dataDict[key].clear() for key in self.dataDict.keys() ]
