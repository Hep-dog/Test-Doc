[文档]def getSecAllCounting( dataDict, batchSize ):
    secCounting = []
    secIDList = []
    signalList   = dataDict["leftData"]   + dataDict["dataList"]
    signalIDList = dataDict["leftIDList"] + dataDict["idList"]
    dataDict["leftData"].clear()
    dataDict["leftIDList"].clear()

    for i in range(len(signalList)%batchSize):
        dataDict["leftData"]  .insert( 0, signalList.pop() )
        dataDict["leftIDList"].insert( 0, signalIDList.pop() )
    if len(signalList)/batchSize>=1:
        secCounting = list( np.sum( (np.array(signalList)).reshape(-1,batchSize), axis=1  ) )
        secIDList   = (np.array(signalIDList).reshape(-1,batchSize))[:,0].tolist()
    return secCounting, secIDList
