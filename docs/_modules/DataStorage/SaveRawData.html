    """
        Inherited from the template data process class "MsgProcess"
        This class is used to save the CSNS raw data, to the original binary format.\n
        Note: the corresponding binary data should be decoded with the data structure from
        electronics.
    """
    def __init__(self, appName):
        super().__init__(appName)
        self.subDirName = "RawData"
        self.dataLength = 10000
        self.dataFile   = None
        self.dataPath   = "output/"
        self.dataFileName = ""
        self.fileProcessCounter = 0

[文档]    def createBinFile( self, msg, ID ):
        """
            Creating the files
        """
        # Create the data file firstly
        # Check the data path is valid or not
        dataDir = self.dataPath + "/" + msg.topic + "/" + self.subDirName + "/"
        if not os.path.exists( dataDir ):
            os.makedirs( dataDir )

        self.dataFileName = dataDir + msg.topic + "-ID" + str( ID ) + ".bin"
        self.dataFile = open(str( self.dataFileName ), "wb")
        print("\033[42mINFO: Create data Bin file: %s\033[0m" % self.dataFileName )



[文档]    def saveBinData( self, RawVec, cnt ):
        """
            Do the data storage.
        """
        self.dataFile.write( RawVec )
        self.fileProcessCounter += 1
        if self.fileProcessCounter == ( self.dataLength ):
            self.dataFile.close()
            self.fileProcessCounter = 0
