        """
            Renaming the finished ROOT file.
        """
        oldName = self.outputName
        joinID  = str(self.headID) + "-" + str(self.tailID)
        newName = self.outputName.replace( str(self.headID), joinID )
        cmd = "mv " + oldName + "  " + newName
        os.system( cmd )
        self.outputName = newName

[文档]    def createROOTFile( self, msg ):
        """
            Creating the ROOT file, tree and renaming the variables.
        """
        dataDir = self.outputDir + msg.topic + "/"
        if not os.path.exists( dataDir ):
            os.makedirs( dataDir )
        self.outputName = dataDir + "ID-" + str(self.headID) + ".root"
        try:
            self.outputFile = ROOT.TFile( self.outputName, "RECREATE" )
        except:
            traceback.print_exc()
        self.outputTree = ROOT.TTree( "EventData", "" )
        self.pulseID  = np.zeros( 1, dtype=np.int64 )
        self.eventLen = np.zeros( 1, dtype=np.int32 )
        self.channel  = np.zeros( 1, dtype=np.int32 )
        self.tof      = np.zeros( 1, dtype=np.int64 )
        self.eventNum = np.zeros( 1, dtype=np.int64 )

        self.outputTree.Branch( "tof"     , self.tof     , "tof/L" )
        self.outputTree.Branch( "pulseID" , self.pulseID , "pulseID/L" )
        self.outputTree.Branch( "eventNum", self.eventNum, "eventNum/L" )
        self.outputTree.Branch( "eventLen", self.eventLen, "eventLen/I" )
        self.outputTree.Branch( "channel" , self.channel , "channel/I" )



[文档]    def closeROOTFile(self):
        self.outputFile.Write()
        self.outputFile.Close()



[文档]    def run( self, msg, eventCounter, detectorTypeCode, dataType, dataPipeDict ):
        buf = msg.value
        data = EventData.EventData.GetRootAsEventData( buf, 0 )
        ID = data.PulseId()
        channelVec = data.PosAsNumpy()
        tDataVec   = data.TofAsNumpy()
        eventLen   = data.TofLength()

        # Create the ROOT file and tree
        if (eventCounter) % self.batchSize == 1:
            self.headID = ID
            self.createROOTFile( msg )
            print("\033[42mINFO: Create new ROOT file for: %s\033[0m" % msg.topic )

        # Fill the events of each Pulse to tree
        for i in range(eventLen):
            self.pulseID[0]  = ID
            self.eventLen[0] = eventLen
            self.eventNum[0] += 1
            self.channel[0]  = channelVec[i]
            self.tof[0]      = tDataVec[i]
            self.outputTree.Fill()

        # Closing the root file when reaching the event batch size
        if (eventCounter) % self.batchSize == 0:
            self.tailID = ID
            self.renameROOTFile(msg)
            self.closeROOTFile()
            print( "    ++++++++++++++++++++++ ROOT file has been closed      +++++++++++++    " )
            print("\033[42mINFO: ROOT file has been closed: %s\033[0m" % msg.topic )
        # ======    ======================================================================

        if (eventCounter)%10000 == 1:
            print( msg.topic, ID )
            print( " "*10, "Topic: %46s,\t ID: %10d,\t Offset: %8d "%(msg.topic, ID, msg.offset) )
