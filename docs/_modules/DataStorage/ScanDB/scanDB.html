[文档]def pullValueSum( topic, db, cursor ):

    errorSum=0
    name = topic.replace("-", "_")
    sql = "select * from ID_Offset_Status_" + name + " order by DataID desc limit 1000"
    #print(sql)
    cursor.execute( sql )
    result = cursor.fetchall()

    for msg in result:
        errorSum += list(msg)[3]
        #print( errorSum, list(msg)[3],  )

    return errorSum
