[文档]def timeToID( standardTime:str ) -> int:
    """
    Converting the timestamp the expected pulseID.

    Args:
        standardTime : time in formate "%Y-%m-%d %H:%M:%S"
    """
    global G_BASIC_ABSTIMESTAMP
    global G_BASIC_ABSPULSEID
    tarray = time.strptime( standardTime, "%Y-%m-%d %H:%M:%S" )
    stamp  = int( time.mktime(tarray) )
    expectedID = (stamp-G_BASIC_ABSTIMESTAMP)*25 + G_BASIC_ABSPULSEID
    return expectedID



[文档]def collectData( resultList, endPoint, Metric, value, time_st ):
    """
        Template function used to format data for further push to OpenFalcon platform\n
        Args:\n
            resultList : list holds records of user target visualization results\n
            endPoint (str) : name of endPoint for OpenFalcon\n
            Metric (str) : name of metric for OpenFalcon\n

            time_st      : time stamp for OpenFalcon\n
    """
    record = {}
    record['endPoint']      = endPoint
    record['step']          = 30
    record['counterType']   = 'GAUGE'
    record['metric']        = Metric
    record['value']         = value
    record['timestamp']     = time_st
    resultList.append( record )

[文档]def monitorDevideProton( v_moni:int, v_proton:int ) -> float:
    """
        Event rate of Monitor divides the one from proton beam.

        Args:\n
            v_moni (int) : event rate of monitor\n
            v_proton (int) : event rate of proton beam\n

        Return: float
    """
    try:
        if v_proton>0:
            return v_moni/ v_proton
        elif v_proton == 0.0:
            return 0.0
    except:
        return -1

[文档]def IDToTime( pulseID ):
    """
        Converting the pulse id to time in format, "%Y-%m-%d %H:%M:%S"
    """
    global G_PULSEFRE
    absID   = 451021798
    absStamp = 1618901432 # 2021-04-20 14:50:32
    try:
        expStamp = absStamp + (pulseID - absID)/G_PULSEFRE
        humanTime = time.strftime("%Y-%m-%d %H:%M:%S", time.localtime(expStamp))
        return humanTime
    except:
        traceback.print_exc()
        return None




[文档]def getSecAllCounting( dataList, idList, leftDataList, leftIDList, pulseFre ):
    """
        This function used to add the second to column of the aggregated dataframe results,
        which is used to obtain the total counting rate of all detector modules.

        Args:\n
            dataList : List of the aggregated dataframe results ( 1-end columns )\n
            idList   : List of the ID column of the aggregated dataframe results ( 0 column )\n
            leftDataList : left List of the aggregated dataframe results from last aggregation\n
            leftIDList   : left List of the pulse ID of aggregated dataframe results from last aggregation\n
            pulseFre     : CSNS pulse frequency

        Return:
            List of event rate results and the corresponding list of pulse ID
    """
    secCounting = []
    secIDList = []
    signalList = leftDataList + dataList
    signalIDList = leftIDList + idList
    leftDataList.clear()
    leftIDList.clear()

    for i in range(len(signalList)%pulseFre):
        leftDataList.insert( 0, signalList.pop() )
        leftIDList.insert( 0, signalIDList.pop() )
    if len(signalList)/pulseFre>=1:
        secCounting = list( np.sum( (np.array(signalList)).reshape(-1,pulseFre), axis=1  ) )
        secIDList   = (np.array(signalIDList).reshape(-1,pulseFre))[:,0].tolist()
    return secCounting, secIDList



[文档]def onlineVisualAll( resultList, endPoint, metric ):
    """
        Standard function displays the list of results.
    """
    targetList = []
    time_st = int(time.time())
    for item in resultList:
        collectData( targetList, endPoint, metric, item, time_st )

    roll = requestPV(targetList)
    print( " ===================== New round ========================= " )
    print( " EndPoint: %s, NameMetric: %s, Counts: %s" % ( endPoint, metric, resultList  )  )
    print( " ++++++++++++++++++++++++++++ requests sending status: ", roll.text, " ++++++++++++++++++++++++ "  )



[文档]def onlineVisualSingle( resultList, endPoint, metric ):
    """
        Function used to display single result.
    """
    transDict = {}
    transDict['mean'] = np.mean(resultList)
    transDict['max']  = np.max(resultList)
    transDict['min']  = np.min(resultList)

    targetList = []
    time_st = int(time.time())
    for key in transDict:
        temptMetric = metric + '-' + key
        collectData( targetList, endPoint, temptMetric, transDict[key], time_st )

    roll = requestPV(targetList)
    print( " ===================== New round ========================= " )
    print( " EndPoint: %s, NameMetric: %s, Counts: %s" % ( endPoint, metric, transDict  )  )
    print( " ++++++++++++++++++++++++++++ requests sending status: ", roll.text, " ++++++++++++++++++++++++ "  )
