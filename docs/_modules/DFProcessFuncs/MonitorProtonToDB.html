[文档]    def processDF( self, df ):
        self.dataList = df.values.tolist()
        self.idList   = df.index.values.tolist()

        for i in range(len(self.idList)):
            try:
                pulseID, dataVec = self.idList[i], self.dataList[i]
                targetList = [ pulseID, dataVec[0], dataVec[1][0], dataVec[1][1][0], dataVec[1][2][0] ]
                self.writeToDB( targetList )
            except:
                traceback.print_exc()
                print( "\033[41mERROR: Facing Error in the signal dataframe processing!\033[0m" )
                sys.exit(1)

        if len(self.idList)>0:
            print( " IDs of successfully written to database are: %s - %s " % (self.idList[0], self.idList[-1] ) )
