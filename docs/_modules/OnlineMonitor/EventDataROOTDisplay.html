    def __init__(self, appName):
        super().__init__(appName)
        self.batchSize      = 1500
        self.detectorType   = 0 # 1 for He3, 2 for Monitor, 3 for Scin,
        self.subdetectorType= 0 # He3: 0 for long tube and 1 for short, Scin: 0 and 1 are same,
        self.pixelRange     = ( (100,60), (32,32), (32,32) ) # long tube: 100*8, short tube: 60*8
        self.tofFactor      = ( 1, 1, 8 ) # 1 for He3 and Monitor, 8 for Scin
        self.tofUnit        = ( 1000, 1000, 1000000 ) # ms for He3 and Monitor, ns for Scin
        self.canvas1D       = None
        self.canvas2D       = None
        self.hbatchTof      = None
        self.htotalTof      = None
        self.hbatchPixel    = None
        self.htotalPixel    = None
        self.hDelPulseID    = None
        self.hCounting      = None
        self.prePulseID     = -1
        self.MPI_ENCODE_TAIL= 10000 # valid digital for MPI detector pixel number
        self.tofBins        = 400
        self.pixelBinFactor = 1

[文档]    def getPixelX( self, pixel ):
        """
            Transferring the 1D digit pixel data to 2D detector pixel.\n
            i.e. 1-800 (long 3He tube) should be (x,y): (1-100, 0-7).
        """
        res = pixel%100
        if res==0:
            return 100
        else:
            return res

[文档]    def setDetectorType( self, topicName, detectorTypeCode ):
        """
            Getting the detector type with input detectorTypeCode, which is used to
            decode the pixel data of EventData
        """
        if int(detectorTypeCode)/100000<0 or int(detectorTypeCode)/100000>10:
            print( "\033[41mERROR: Topic: %s input Detector Type Code, exist the program!\033[0m" % topicName )
            sys.exit(1)
        self.detectorType = int(int(detectorTypeCode)/100000)-1
        if self.detectorType == 0 or self.detectorType==1:
            self.subdetectorType= int( (int(detectorTypeCode)%10) )
        print( "\033[42mINFO: Topic: %s Detector type: %s, subdetectorType: %s, pixel Range: %s\033[0m" \
            % ( topicName, self.detectorType, self.subdetectorType, self.pixelRange[self.detectorType][self.subdetectorType] ) )



[文档]    def initCanvasHists(self):
        """
            Initializing the ROOT histogram
        """
        self.canvas1D = ROOT.TCanvas( "canvas1D", "Canvas 1D", 1000, 800 )
        self.canvas2D = ROOT.TCanvas( "canvas2D", "Canvas 2D", 1000, 800 )
        self.canvas1D.Divide(2,2)
        if self.detectorType ==2:
            self.canvas2D.Divide(2,1)
        else:
            self.canvas2D.Divide(1,2)
        self.hbatchTof = ROOT.TH1F( "hbatchTof", "", self.tofBins, 0, 40 )
        self.htotalTof = ROOT.TH1F( "htotalTof", "", self.tofBins, 0, 40 )
        self.hDelPulseID = ROOT.TH1F( "hDelPulseID", "", int(self.batchSize), 0, self.batchSize )
        self.hCounting   = ROOT.TH1F( "Counting", "", int(self.batchSize), 0, self.batchSize )

        #self.hDelPulseID = ROOT.TH2F( "hDelPulseID", "", 150, 0, self.batchSize, 10, -5, 5 )
        if self.detectorType == 2:
            self.hbatchPos = ROOT.TH1F( "hbatchPos", "", \
                    self.pixelRange[self.detectorType][self.subdetectorType], \
                    0,
                    self.pixelRange[self.detectorType][self.subdetectorType] )
            self.htotalPos = ROOT.TH1F( "htotalPos", "", \
                    self.pixelRange[self.detectorType][self.subdetectorType], \
                    0,
                    self.pixelRange[self.detectorType][self.subdetectorType] )
        else:
            self.hbatchPos = ROOT.TH2F( "hbatchPos", "", \
                    self.pixelRange[self.detectorType][self.subdetectorType]*self.pixelBinFactor, \
                    0,
                    self.pixelRange[self.detectorType][self.subdetectorType], \
                    10, 0, 10)
            self.htotalPos = ROOT.TH2F( "htotalPos", "", \
                    self.pixelRange[self.detectorType][self.subdetectorType]*self.pixelBinFactor, \
                    0,
                    self.pixelRange[self.detectorType][self.subdetectorType], \
                    10, 0, 10)
        # SetTitle
        self.setHistTitle( self.htotalTof, "Accumulated Events", "Tof (ms)" )
        self.setHistTitle( self.hbatchTof, "Current Events"    , "Tof (ms)" )
        self.setHistTitle( self.hDelPulseID, "#{Delta}Pulse ID", "last " + str(self.batchSize) + " pulses" )
        self.setHistTitle( self.hCounting , "Counting Rate", "last " + str(self.batchSize) + " pulses" )
        self.setHistTitle( self.hbatchPos , "Current Pixel X ranges", "Current Pixel Y ranges")
        self.setHistTitle( self.htotalPos , "Accumulated Pixel X ranges", "Accumulated Pixel Y ranges")

[文档]    def run( self, msg, eventCounter, detectorTypeCode, dataType, dataPipeDict ):
        buf = msg.value
        data = EventData.EventData.GetRootAsEventData( buf, 0 )
        ID = data.PulseId()
        TofArray = data.TofAsNumpy()
        PosArray = data.PosAsNumpy()

        if eventCounter == 1:
            self.prePulseID = ID
        #print( "========================= Pre Pulse ID is: %d, %d  ==================== " % ( self.prePulseID, ID-self.prePulseID) )
        if eventCounter % self.batchSize != 0:
            self.hDelPulseID.SetBinContent( eventCounter%self.batchSize, ID-self.prePulseID )
            self.hCounting.SetBinContent( eventCounter%self.batchSize, data.TofLength() )
            #self.hDelPulseID.Fill( int(eventCounter%self.batchSize), int(ID-self.prePulseID) )
        else:
            self.hDelPulseID.SetBinContent( self.batchSize, ID-self.prePulseID )
            self.hCounting.SetBinContent( self.batchSize, data.TofLength() )
            #self.hDelPulseID.Fill( int(self.batchSize), int(ID-self.prePulseID) )

        for i in range(len(TofArray)):
            self.htotalTof.Fill( TofArray[i] * self.tofFactor[self.detectorType] \
                                                        / self.tofUnit[self.detectorType] )
            self.hbatchTof.Fill( TofArray[i] * self.tofFactor[self.detectorType] \
                                                        / self.tofUnit[self.detectorType] )
            if self.detectorType == 2:
                self.htotalPos.Fill( PosArray[i] )
                self.hbatchPos.Fill( PosArray[i] )
            else:
                transPixel = PosArray[i]%self.MPI_ENCODE_TAIL
                curX = self.getPixelX( transPixel )
                curY = int( transPixel/100 ) + 1
                self.hbatchPos.Fill( curX, curY)
                self.htotalPos.Fill( curX, curY)


        if eventCounter%self.batchSize ==0:
            self.canvas1D.cd(1)
            self.htotalTof.Draw()
            self.canvas1D.cd(2)
            self.hbatchTof.Draw()
            self.canvas1D.cd(3)
            self.hDelPulseID.Draw()
            self.canvas1D.cd(4)
            self.hCounting.Draw()

            if self.detectorType ==2:
                self.canvas2D.cd(1)
                self.hbatchPos.Draw()
                self.canvas2D.cd(2)
                self.htotalPos.Draw()
            else:
                self.canvas2D.cd(1)
                self.hbatchPos.Draw("colz")
                self.canvas2D.cd(2)
                self.htotalPos.Draw("colz")

            self.canvas1D.Modified()
            self.canvas1D.Update()
            self.canvas2D.Modified()
            self.canvas2D.Update()

            self.hbatchTof.Reset()
            self.hbatchPos.Reset()

        self.prePulseID = ID

        if eventCounter%10000 == 1:
            print( " "*10, "Topic: %45s,\t ID: %10d,\t Offset: %8d "%(msg.topic, ID, msg.offset) )
