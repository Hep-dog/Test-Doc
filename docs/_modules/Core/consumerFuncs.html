[文档]def timeToID( standardTime:str, absTimeStamp:int, absPulseID:int ) -> int:
    """
        Converting the timestamp the CSNS pulseID\n
        Args:\n
            standardTime(str) : the target time (format: "%Y-%m-%d %H:%M:%S")\n
            absTimeStamp(int) : the absolute time stamp of specified CSNS pulse\n
            absPulseID(int)   : the absolute pulse ID of specified CSNS pulse\n
        Return:\n
            int: the expected pulse ID for the input time
    """
    tarray = time.strptime( standardTime, "%Y-%m-%d %H:%M:%S" )
    stamp  = int( time.mktime(tarray) )
    expectedID = (stamp-absTimeStamp)*25 + absPulseID
    return expectedID




[文档]def expOffsetFromID( consumer, partion, numb, protLength, jumpPulseID, absTimeStamp, absPulseID ) -> int:
    """
        Used to convert the input jumping Pulse ID to the expected Kafka offset of each partition for target topic\n
        Agrs:\n
            consumer : Kafka consumer instance\n
            partion  : partition of the topic\n
            numb     : total number of patitions for target topic\n
            protLength : input jumping protecting message length\n
            jumpPulseID: target jumping pulse ID\n
            absTimeStamp : the absolute time stamp of specified CSNS pulse\n
            absPulseID(int)   : the absolute pulse ID of specified CSNS pulse\n
        Return:\n
            int : expected offset of the jumping pulse ID for the partition\n
    """
    deltaID   = absPulseID - jumpPulseID + int(protLength*numb)
    expTime = absTimeStamp - deltaID/25.
    dictStamp = { partion:expTime*1000 }
    T2Offset  = consumer.offsets_for_times( dictStamp )[partion]
    #print( deltaID, expTime, absTimeStamp, dictStamp, T2Offset )
    #print( " ============== The expected offset from ID is: %s ============ " % T2Offset.offset )
    return T2Offset.offset




[文档]def createConsumers( IP:str, nameTopic:str, Group_ID:str, numb:int, Consumers:List[str], consumerParasDict:Dict[str,str] ) -> None:
    '''
        Used to create multi-consumers and return the consumers list, each one for one partition of a specified topic.\n
        Args:\n
            IP: kafka broker address and port\n
            nameTopic: Kafka topic name\n
            Group_ID : name of consumer group\n
            numb: number of partitions of the topic\n
            consuemrParasDict: input user consuming mode configurations file\n
            Consumers: list use to hold all consumers instances\n
            jumpPulseID (in consumerParasDict): targe position user actually consuming for each partitions, define the expected jump offset:\n
                0) from the lastest real-time msg if jumpPulseID is '-2', which measn "group_id" doesn't works\n
                1) from the latest msg if jumpPulseID is '-1' (continue with offset position from the last data consuming)\n
                2) from the beginning for '0' if jumpPulseID is 0\n
                3) from a specific position if expOffset is an integer larger than 0.\n

    Return: None
    '''
    TPs = [] # list of topicpartitions
    expOffsets  = [] # list of expected offset for each partition
    timeout_ms  = consumerParasDict['timeout_ms']
    protLength  = consumerParasDict['protLength']
    jumpPulseID = consumerParasDict['jumpPulseID']
    absPulseID  = consumerParasDict['absPulseID']
    absTimeStamp= consumerParasDict['absTimeStamp']

    for i in range(numb):
        consumer = KafkaConsumer(
            bootstrap_servers = [IP],
            group_id = Group_ID,
            consumer_timeout_ms = timeout_ms,
        )
        tp = TopicPartition( nameTopic, i )
        consumer.assign( [tp] )
        TPs.append( tp )
        Consumers.append( consumer )

        if consumerParasDict['filterGate']:
            if jumpPulseID == -2:
                try:
                    realTimeOffset =  consumer.end_offsets([tp])[tp]
                    consumer.seek( tp, realTimeOffset )
                    print("\033[44mINFO: %50s's partition: %d consumes the real-time message from offset: %d" % (nameTopic, i, realTimeOffset), "\033[0m" )
                except:
                    print("\033[41mERROR: Topic: %50s seeks to the real-time message error!\033[0m" % nameTopic)
                    traceback.print_exc()
                    sys.exit(1)

            elif jumpPulseID == -1:
                print("\033[44mINFO: %50s's partition: %d consumes from the latest offset committed position " % (nameTopic, i), "\033[0m" )
                pass

            elif jumpPulseID == 0:
                #Consumers[i].seek( TPs[i], 0 )
                Consumers[i].seek_to_beginning( TPs[i] )
                print("\033[44mINFO: %50s's partition: %d consumes from beginning " % (nameTopic, i), "\033[0m" )

            else:
                temptOffset = expOffsetFromID( consumer, tp, numb, protLength, jumpPulseID, absTimeStamp, absPulseID )
                expOffsets.append( temptOffset )
                if expOffsets[i]>0:
                    try:
                        consumer.seek( tp, (expOffsets[i]-1) )
                        print("\033[44mINFO: %50s's partition: %d consumes from offset: %d " % (nameTopic, i, expOffsets[i]), "\033[0m" )

                    except:
                        traceback.print_exc()
                        exit(1)
                else:
                    try:
                        consumer.seek( tp, 0 )
                    except:
                        traceback.print_exc()
                        exit(1)




[文档]def getID( msg:bytes, Invalid_ID:int ) -> int:
    '''
        Used to get the pulse ID for input message.\n
        The decoding method is automatically generated by flatbuffers serialization.\n
        Note: since we read messages from all partions, there will be a case that some partitions have messages and others are "None".
        For this case, we set the invalid_ID for the pulse ID for the "None" messages. So the "None" messages will be in the end
        of the messages pool by sorting with the pulse ID, we just left the "None" messages in the pool and don't use them.\n
        Args:\n
            msg : recieved msg data ( in binary format )\n
            Invalid_ID : the defined invalid pulse ID to identify tue non-physical data\n
        Return:
            the pulse ID
    '''
    if msg:
        buf  = msg.value
        data = EventData.EventData.GetRootAsEventData( buf, 0 )
        ID   = data.PulseId()
        return ID
    else:
        return ( Invalid_ID )

[文档]def fillPools( Pool_msg:List[bytes], Pool_ID:List[int], msg:bytes, Invalid_ID:int ) -> None:
    '''
        Note: the messges are sorted by its key, whose value is same as the Pulse ID.
        So, the sorting procedure is independent on the flatbuffers encoding/decoding method.

        Args:\n
            Pool_msg   :  global messages pool
            Pool_ID    :  global Pulse ID pool
            msg        :  messages data
            Invalid_ID : invalid pulse ID for the 'None' messages
    '''
    Pool_msg.append( msg )

    if msg:
        #ID = (struct.unpack('<2L', msg.key))[0]
        ID = getID( msg, Invalid_ID )
        Pool_ID.append( ID )
    else:
        Pool_ID.append( Invalid_ID )



[文档]def freeWater( Pool_msg, Pool_ID, sliceSize, Last_msg, Last_ID, sortFlag, filterGate):
    '''
        Used to release messages and IDs from the pools when their are full.\n
        Args:\n
            Pool_msg: the fulled messages pool\n
            Pool_ID : the filled IDs pool\n
            waterList: the target sorted messages list, which will be used in further by users\n
            waterIDList: the pulse ID of target sorted messages list, which will be used in further by users\n
            sliceSize: the length of front messages in the filled messages pool\n
            Last_msg: the left messages in the pool after the releasing of water from pool\n
            Last_ID : the left IDs in the pool after the releasing of water from pool\n
            sortFlag: the flag for message sorting or not\n
            filterGate: the gate for message filtering or not\n
        Return:\n
            List[msg], List[ID], List[msg], List[ID], List[msg], List[ID]
    '''
    leftMSG = []
    leftID  = []
    waterList = []
    waterIDList = []
    #print("TempIDlist length is: ", len(Pool_ID), type(Pool_ID) )

    # If sortflag is True, we sort messages firstly and pull the top sliceSize ones to waterList
    if sortFlag:
        # fill the water list
        tempIDlist = list( np.argsort(Pool_ID) )
        for item in tempIDlist[ 0 : sliceSize ]:
            waterList.append( Pool_msg[item] )
            waterIDList.append( Pool_ID[item] )

        # get the left messages and IDs
        for item in tempIDlist[ sliceSize : ]:
            leftMSG.append( Pool_msg[item] )
            leftID .append( Pool_ID[item] )

    # If we don't need to sort the messages, just release all messages to waterList
    else:
        waterList = Pool_msg
        waterIDList = Pool_ID

    Pool_msg = leftMSG
    Pool_ID  = leftID

    Last_msg = leftMSG
    Last_ID  = leftID

    return waterList, waterIDList, Pool_msg, Pool_ID, Last_msg, Last_ID




[文档]def filterWater( waterList, waterIDList, consumerParasDict ):
    '''
        This function is used to filter the messages from user specified Pulse ID.
        Normally, the first message ID will be not the target jumping pulse ID,
        so we need to use this function to filter our wanted messages.\n

        Agrs:\n
            waterList: the target sorted messages list, which will be used in further by user.\n
            waterIDList: the IDs list of waterList, which is used for the messages filtering\n
            consuemrParasDict: input user consuming mode configurations file\n
        Return:
            List[msg], List[ID], bool( msg filtering control flag )
    '''
    sortFlag = consumerParasDict['sortFlag']
    poolSize = consumerParasDict['poolSize']
    filterGate = consumerParasDict['filterGate']
    Invalid_ID = consumerParasDict['Invalid_ID']
    jumpPulseID = consumerParasDict['jumpPulseID']
    tempWaterList = []
    tempIDList = []

    if len(waterList) != len(waterIDList):
        print("\033[41mERROR: the length of water list and waterID list are not equal during the filtering!\033[0m")
        sys.exit(1)

    if sortFlag:
        for msg, ID in zip( waterList, waterIDList ):
            if ID >= jumpPulseID and ID < Invalid_ID:
                tempWaterList.append( msg )
                tempIDList.append( ID )
                filterGate = False
    else:
        for msg, ID in zip( waterList, waterIDList ):
            if ID >= ( jumpPulseID ) and ID < Invalid_ID:
                tempWaterList.append( msg )
                tempIDList.append( ID )
            if ID >= ( jumpPulseID + poolSize ):
                filterGate = False


    waterList   = tempWaterList
    waterIDList = tempIDList

    #print( "Two length in filterWater function : ", len(waterList), len(waterIDList) )
    return waterList, waterIDList, filterGate

    '''
        Used to cut the messages to user specified pulse ID, if cutoffFlag is true.\n
        Args:\n
            waterList  : the input signal messages list, which has be sorted in most cases.\n
            waterIDList: the input message IDs list, its length should be same as waterList.\n
            cutoffFlag : the control flag used to decide cut messages or not.\n
            endPulseID : user specified end of messages.\n
        Return:
            List[msg], List[PulseID]
    '''
    if not cutoffFlag:
        return waterList, waterIDList
    else:
        if endPulseID <= 0:
            print("\033[41mERROR: the given endPulseID is invalid, please check it!\033[0m")
            exit(1)
        else:
            if len(waterList) != len(waterIDList):
                #print( "Two length: ", len(waterList), len(waterIDList) )
                print("\033[41mERROR: the length of water list and waterID list are not equal after the message filtering!\033[0m")
                sys.exit(1)
            else:
                MSGList = []
                IDList  = []
                for msg, ID in zip( waterList, waterIDList ):
                    if ID <= endPulseID:
                        MSGList.append( msg )
                        IDList .append( ID  )
                    else:
                        exit(0)
                return MSGList, IDList




[文档]def releasePool( eventLoop, executor, detectorTypeCode, dataType, Last_msg, Last_ID, consumerParasDict, glbCtrlDict, eventCounter, dataPipeDict, userObjectList ) -> None:
    '''
        Used to release the message pool after the final loop, since after the last message loop,
        the message pool will be not full, which means the size of Last_msg and Last_ID will smaller
        than the poolSize. And there are must be some "None" messages in the Pool_msg and invalid_ID
        in the Pool_ID, because the number of messages of each partition are not same. For this case,
        we use the addtional "None" messages to full the Pool_msg and Pool_ID, sort the messages and pull the valid ones.\n
        Args:\n
            eventLoop: asyncio event loop\n
            executor : asyncio executor\n
            detectorTypeCode : the encoding of detector type\n
            Last_msg : list hold the msg after free water function\n
            Last_ID  : list hold the IDs after free water function\n
            consuemrParasDict: input user consuming mode configurations file\n
            glbCtrlDict : global program running ctrl dict\n
            eventCounter: the global event counter for each topic\n
            dataPipeDict: data pipe dict used to transfer the data after user processing, from this process to aggregation process\n
            userObjectList: list hold user data processing algorithms;
    '''
    if len(Last_msg) != len(Last_ID):
        print( "\033[41mERROR: The number of the left msg %d and ID %d are not same!" % ( len(Last_msg), len(Last_ID) ), "\033[0m" )
        sys.exit(1)

    leftSize = len(Last_ID)
    blankSize= consumerParasDict['poolSize'] - leftSize
    mudSize  = Last_ID.count( consumerParasDict['Invalid_ID'] )
    dataSize = leftSize - mudSize

    for i in range( blankSize ):
        Last_msg.append( None )
        Last_ID .append( consumerParasDict['Invalid_ID'] )

    tempIDlist = list( np.argsort(Last_ID) )

    for item in tempIDlist[ 0 : dataSize ]:
        '''
            User should overload this function or re-write it!!!!!
        '''
        #ID = (struct.unpack('<2L', (Last_msg[item]).key))[0]
        ID = getID( Last_msg[item], consumerParasDict['Invalid_ID'] )

        # For message jumping and cutting off flags
        if consumerParasDict['filterGate'] and (ID < consumerParasDict['jumpPulseID']):
            continue
        elif consumerParasDict['cutoffFlag'] and (ID > consumerParasDict['endPulseID']):
            continue
        else:
            msgRun( consumerParasDict['coroutineFlag'], eventLoop, executor, detectorTypeCode, dataType, Last_msg[item], consumerParasDict['batchModeFlag'], consumerParasDict['runStep'],\
                   consumerParasDict["slowConsumerInSecond"], glbCtrlDict, eventCounter, dataPipeDict, userObjectList, )

[文档]def msgRun( coroutineFlag, eventLoop, executor, detectorTypeCode, dataType, msg, batchModeFlag, runStep, slowConsumerInSecond, glbCtrlDict, eventCounter, dataPipeDict, userObjectList ):
    """
        User message processing interface, which is the entrance for user's data processing codes\n
        Args:\n
            coroutineFlag : asyncio coroutineFlag mode control flag\n
            eventLoop : asyncio event loop\n
            executor  : asyncio executor\n
            detectorTypeCode : the encoding of detector type\n
            dataType  : data types of CSNS data format, should be "EventData", "RawData" or "MetaData"\n
            msg : signal data streaming after the pre data processing\n
            batchModeFlag : input control flag for synchronization consuming of multiple topics, True or False\n
            runStep       : run step of batch consuming mode\n
            slowConsumerInSecond : input factor in second to slow down user consuming speed\n
            glbCtrlDict : global program running ctrl dict\n
            eventCounter: the global event counter for each topic\n
            dataPipeDict: data pipe dict used to transfer the data after user processing, from this process to aggregation process\n
            userObjectList: list hold user data processing algorithms\n
        Return:\n
            int : eventCounter
    """
    if msg:
        fixCPUTime( slowConsumerInSecond )
        if not batchModeFlag:
            eventCounter += 1
            ## TODO
            taskList = []
            if coroutineFlag:
                for i in range(len(userObjectList)):
                    taskList.append( asynDataProcess( eventLoop, executor, msg, eventCounter, detectorTypeCode, dataType, dataPipeDict, userObjectList[i] ) )
                eventLoop.run_until_complete( asyncio.gather(*taskList) )
            else:
                for i in range(len(userObjectList)):
                    userObjectList[i].run( msg, eventCounter, detectorTypeCode, dataType, dataPipeDict  )
        else:
            eventCounter += 1
            if not glbCtrlDict['event'].is_set():
                if( eventCounter % runStep == 0 ):
                    print("\033[46mINFO: Step msgRun running ! \033[0m", (eventCounter , glbCtrlDict['progressCNT'].value) )
                    glbCtrlDict['lock'].acquire()
                    glbCtrlDict['progressCNT'].value += 1
                    glbCtrlDict['lock'].release()
                    glbCtrlDict['event'].wait()
            else:
                if( eventCounter % runStep == 0 ):
                    print("\033[46mINFO: Step msgRun running ! \033[0m", (eventCounter , glbCtrlDict['progressCNT'].value) )
                    glbCtrlDict['lock'].acquire()
                    glbCtrlDict['progressCNT'].value += 1
                    glbCtrlDict['event'].clear()
                    glbCtrlDict['lock'].release()
                    glbCtrlDict['event'].wait()

            ## TODO
            taskList = []
            if coroutineFlag:
                for i in range(len(userObjectList)):
                    taskList.append( asynDataProcess( eventLoop, executor, msg, eventCounter, detectorTypeCode, dataType, dataPipeDict, userObjectList[i] ) )
                eventLoop.run_until_complete( asyncio.gather(*taskList) )
            else:
                for i in range(len(userObjectList)):
                    userObjectList[i].run( msg, eventCounter, detectorTypeCode, dataType, dataPipeDict  )
    else:
        print("\033[42mWARNING: facing the 'None' message!\033[0m")
    return eventCounter




[文档]def fixCPUTime( waitTime:int ) -> None:
    '''
        Used to occupy a fixed CPU calculation time. The 'time.sleep(t)' method will
        pause this thread/process to other threads/processes, so the actual time used
        will be large than the target time.\n
        Args:\n
            waitTime (int) : input wating time\n
        Retrun:
            None
    '''
    start = time.process_time()
    while True:
        if (time.process_time()-start) > waitTime:
            break
